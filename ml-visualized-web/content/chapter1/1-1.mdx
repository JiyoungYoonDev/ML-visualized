<Block

</Block>

<Block
  label='Algorithm Flow'

    <li>
      compare weighted votes
      <ul className='mt-1 list-disc pl-5'>
        <li>
          label 1: <LatexMath math={'\\sum_{i:x_i(t)=1}w_i(t)'} />

        </li>
      </ul>
    </li>
    <li>learner predicts majority-by-weight</li>
    <li>
      observe <LatexMath math={'y(t)'} />
    </li>

      penalize wrong experts:{' '}
